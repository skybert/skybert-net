#+title: JRebel 

To start JBoss,I use:
#+begin_src sh
workspace_dir=/home/torstein/src/workspace-eclim
jrebel_jar=/opt/jrebel/jrebel.jar
export JREBEL_OPTS="
  -javaagent:${jrebel_jar}
  -Drebel.workspace.path=${workspace_dir}
  -Drebel.log.file=$HOME/.jrebel/jrebel.log
  -Drebel.properties=$HOME/.jrebel/jrebel.properties
"

$(dirname $0)/standalone.sh $@
#+end_src

To configure JRebel, I use:
#+begin_src sh
rebel.license.url=https://example.com/key

rebel.log=true
# For some reason, this is always relative to $HOME, 
rebel.log.file=.jrebel/jrebel.log

rebel.properties.version=2
rebel.license=$HOME/.jrebel/client.lic

# since I'm runnign jrebel as a different user, I don't use $HOME here
rebel.workspace.path=/home/torstein/src/workspace-eclim
#+end_src
